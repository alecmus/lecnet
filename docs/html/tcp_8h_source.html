46  //
47  // if (connect()) {
48  // while (connecting()) {
49  // // wait
50  // }
51  //
52  // if (connected()) {
53  // while (running()) {
54  // // use server-client connection
55  // }
56  // }
57  // else {
58  // // connection error
59  // }
60  // }
61  // else {
62  // // connection error
63  // }
64  //
65  // While running() is true, there are two ways of using the server-client connection
66  //
67  // 1. Blocking method (faster)
68  //
69  // auto busy_function = []() {
70  // // wait
71  // return true;
72  // };
73  //
74  // std::string send, received;
75  // if (send_data(send, received, busy_function)) {
76  // // use received data
77  // }
78  // else {
79  // // send/receive error
80  // }
81  //
82  // 2. Non-blocking method (slower)
83  //
84  // std::string send, received;
85  // unsigned long data_id;

87  // while (sending(data_id)) {
88  // // wait
89  // }
90  //

92  // // use received data
