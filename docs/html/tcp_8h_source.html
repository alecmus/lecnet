42 // Correct usage of the liblec::lecnet::tcp::client class is as follows:
43 //
44 // if (connect()) {
45 // while (connecting()) {
46 // // wait
47 // }
48 //
49 // if (connected()) {
50 // while (running()) {
51 // // use server-client connection
52 // }
53 // }
54 // else {
55 // // connection error
56 // }
57 // }
58 // else {
59 // // connection error
60 // }
61 //
62 // While running() is true, there are two ways of using the server-client connection
63 //
64 // 1. Blocking method (faster)
65 //
66 // auto busy_function = []() {
67 // // wait
68 // return true;
69 // };
70 //
71 // std::string send, received;
72 // if (send_data(send, received, busy_function)) {
73 // // use received data
74 // }
75 // else {
76 // // send/receive error
77 // }
78 //
79 // 2. Non-blocking method (slower)
80 //
81 // std::string send, received;
82 // unsigned long data_id;
83 // if (send_data_async(send, received, data_id) {
84 // while (sending(data_id)) {
85 // // wait
86 // }
87 //
88 // if (get_response(data_id, received)) {
89 // // use received data
90 // }
91 // else {
92 // // send/receive error
93 // }
94 // }
95 //

369 // Correct usage of the liblec::lecnet::tcp::server class is as follows:
370 //
371 // 1. Make a new class and inherit from liblec::lecnet::tcp::server_async and
372 // liblec::lecnet::tcp::server_async_ssl. In the inherited classes provide
373 // implementations for the log() and on_receive() virtual functions
374 //
375 // class my_server_class : public liblec::lecnet::tcp::server_async {
376 // public:
377 // my_server_class() {};
378 // ~my_server_class() {};
379 //
380 // void log(const std::string& time_stamp,
381 // const std::string& event) {
382 // // implementation for log member. Below is a simple example
383 // std::cout << time_stamp << "\t" << event << std::endl;
384 // }
385 //
386 // std::string on_receive(const client_address& address,
387 // const std::string& data_received) {
388 // // implementation for on_receive member. Below is a simple example
389 // return data_received; // simple echo server
390 // }
391 // }
392 //
393 // class my_server_class_ssl : public liblec::lecnet::tcp::server_async_ssl {
394 // ...
395 // }
396 //
397 // 2. Declare a pointer to the server base class and use it as follows
398 //
399 // liblec::lecnet::tcp::server* p_server = nullptr;
400 //
401 // bool use_ssl = ...;
402 //
403 // if (use_ssl)
404 // p_server = new my_server_class_ssl();
405 // else
406 // p_server = new my_server_class();
407 //
408 // 3. Proceed as follows
409 //
410 // if (p_server->start()) {
411 // while (p_server->starting()) {
412 // // wait
413 // }
414 //
415 // while (p_server->running()) {
416 // // main server loop
417 // }
418 // }
419 // else {
420 // // error
421 // }
422 //
423 // if (p_server) {
424 // delete p_server;
425 // p_server = nullptr;
426 // }
