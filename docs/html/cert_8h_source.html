26  // Correct usage of the members of liblec::lecnet::cert to make a pair of digital
27  // certificates for use in a tcp server/client ssl connection is as follows:
28  //
29  // 1. Make CA certificate
30  //
31  // liblec::lecnet::cert::certificate ca_cert;
32  // ca_cert.file_name = "ca.crt";
33  //
34  // liblec::lecnet::cert::private_key ca_key;
35  // ca_key.file_name = "ca.crt"; // make it a single file
36  // ca_key.password = "ca_password123";
37  //
38  // if (!liblec::lecnet::cert::gen_rsa_and_cert(ca_key, ca_cert)) {
39  // // error
40  // }
41  //
42  // 2. Make Server certificate request
43  //
44  // liblec::lecnet::cert::certificate_request server_cert_request;
45  // server_cert_request.file_name = "server.csr";
46  //
47  // liblec::lecnet::cert::private_key server_key;
48  // server_key.file_name = "server.key";
49  // server_key.password = "server_password123";
50  //
51  // if (!liblec::lecnet::cert::gen_rsa_and_csr(server_key, server_cert_request)) {
52  // // error
53  // }
54  //
55  // 3. Sign the server certificate request using the CA certificate
56  // (this creates the server certificate)
57  //
58  // if (!liblec::lecnet::cert::sign_csr("ca.crt",
59  // "ca_password123", "server.csr", "server.crt")) {
60  // // error
61  // }
62  //
63  // 4. Append the contents of server.key to server.crt then delete server.key
64  // 5. Delete the server certificate request
65  //
66  // The file ca.crt is used by the tcp client while server.crt together with the password
67  // to the embedded server private key "server_password123" are used by the tcp server.
68  //
